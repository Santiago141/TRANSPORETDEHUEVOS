<script>
  const slides = document.querySelectorAll('.slide');
  const hint = document.getElementById('hint');
  let cur = 0, total = slides.length;

  // Build dots
  const dotsEl = document.getElementById('dots');
  slides.forEach((_, i) => {
    const d = document.createElement('div');
    d.className = 'dot-nav' + (i === 0 ? ' on' : '');
    dotsEl.appendChild(d);
  });
  const dotEls = document.querySelectorAll('.dot-nav');

  function go(n) {
    slides[cur].style.opacity = 0;
    setTimeout(() => {
      slides[cur].style.display = 'none';
      slides[cur].classList.remove('active');
      dotEls[cur].classList.remove('on');
      cur = (n + total) % total;
      slides[cur].style.display = 'flex';
      requestAnimationFrame(() => {
        requestAnimationFrame(() => {
          slides[cur].style.opacity = 1;
          slides[cur].classList.add('active');
          dotEls[cur].classList.add('on');
        });
      });
    }, 300);
    // Hide hint after first click
    hint.style.opacity = 0;
  }

  document.addEventListener('click', () => go(cur + 1));
  document.addEventListener('keydown', e => {
    if (e.key === 'ArrowRight' || e.key === ' ') go(cur + 1);
    if (e.key === 'ArrowLeft') go(cur - 1);
  });
</script>
